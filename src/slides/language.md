## Tagged unions

## Tagged unions
{class:'notitle'}

```typescript
interface Sms {
  kind: 'sms';
  number: string;
}
interface Email {
  kind: 'email';
  address: string;
}

type Destination = Sms | Email;
```

## Unions
{class:'notitle'}

```javascript
function formatRecipient(dest: Destination) {
  switch(dest.kind) {
    case 'sms':
      return \`Via text message to ${dest.number}\`;
    case 'email':
      return \`Via email to ${dest.address}\`;
  }
}
```
